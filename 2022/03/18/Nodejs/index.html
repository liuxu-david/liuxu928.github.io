mysql 执行语句

– 插入列
– INSERT INTO my_db_01.users (username,password) values (‘tony stark’,’098123’)

– 将 id 为 4 的用户密码更新为‘8888’
– UPDATE my_db_01.users set password=’8888’ WHERE id=4
– UPDATE my_db_01.users set password=’admin123’,status=1 WHERE id=2

– 删除表中数据，记得加 where 控制条件
– DELETE FROM my_db_01.users WHERE id=4

– 演示 where 子句的使用
– SELECT _ FROM my_db_01.users WHERE status=1
– SELECT _ FROM my_db_01.users WHERE id>=2
– SELECT _ FROM my_db_01.users WHERE username<>’ls’
– 使用 and
– SELECT _ FROM my_db_01.users WHERE username<>’ls’ AND id<3
– 使用 or
– SELECT * FROM my_db_01.users WHERE username=’zs’ OR status=1

– 使用升序排列 加不加 asc 一样
– SELECT * FROM my_db_01.users order by status

– desc 是降序的方式默认是 asc 是升序
– SELECT _ FROM my_db_01.users order by id desc
– 多范围排序
– SELECT _ FROM my_db_01.users order by status desc,username asc

– 使用 count()统计表中符合条件的数据有几条
– SELECT count() FROM my_db_01.users WHERE status=0

– 使用 as 关键字对列设置别名
– SELECT username as uname FROM my_db_01.users
